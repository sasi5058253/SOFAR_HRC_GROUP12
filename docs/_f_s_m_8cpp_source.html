156  tf2::convert(*goal_orientation, q_goal);
157  tf2::convert(current_pose.orientation, q_orig);
158  q_inv = q_orig;
159  q_inv[3] = - q_orig[3]; //< inverse
160  q_rel = q_goal*q_inv;
161  q_rel.normalize();
162 
163  double r,p,y;
164 
165  tf2::Matrix3x3 mat_orig(q_orig);
166  mat_orig.getRPY(r,p,y);
167  ROS_DEBUG("%s ORIGINALy: %lf", ARM.c_str(), y*180.0/M_PI);
168 
169 
170  tf2::Matrix3x3 mat(q_rel);
171  mat.getRPY(r,p,y);
172 
173  // allow for perpendicular grasps
174 
175  while (abs(y) >= M_PI/4){ y -= sgn<double>(y)*M_PI/2; }
176 
177  ROS_DEBUG("%s AFTER1 RELATIVEy: %lf", ARM.c_str(), y*180.0/M_PI);
178 
179  q_rel.setRPY(r,p,y); // Calculate the new orientation
180  q_goal = q_orig*q_rel;
181 
182  tf2::Matrix3x3 mat_g(q_goal);
183  mat_g.getRPY(r,p,y);
184  ROS_DEBUG("%s GOALy: %lf", ARM.c_str(), y*180.0/M_PI);
185  q_goal.setRPY(r, 0.0, y);

482 bool FSM::stateEvolution(){
483 
484  bool auto_evolve = true;
485 
486  switch (state){
487  case START:
488  {
489  /* This wait is mainly used in the case of rest_pose
490  after a collision is detected
491  */
492  ros::Duration(collision_wait).sleep();
493  FSM::resetCollisionWait();
494  ROS_INFO("BAXTER_FSM_%s: START", ARM.c_str());
495  next_state = FSM::start();
496  // auto_evolve = true; //< IT MUST evolve automatically from start to reach/err
497  }
498  case REACH:
499  {
500  ROS_INFO("BAXTER_FSM_%s: REACH", ARM.c_str());
501  if (block != nullptr){gripperOpen(true);} //< this means it arrived here from an error stateEvolution
502  // eg. collision. Release each "virtually-held" block before starting.
503  next_state = FSM::reachBlock();
504 
505  if (next_state == IDLE){
506  ROS_INFO("BAXTER_FSM_%s: ALL BLOCKS GRASPED FOR NOW (%d)", ARM.c_str(), BAXTER_ATTEMPTS_);
507  break;
508  }
509  if (next_state == ERR){
510  ROS_INFO("BAXTER_FSM_%s: ERROR DURING REACH", ARM.c_str());
511  break;
512  }
513  ROS_INFO("\t BLOCK FOUND: %s", block->getName().c_str());
514  auto_evolve = false;
515 
516  break;
517  }
518  case PICK:
519  {
520  ROS_INFO("BAXTER_FSM_%s: PICK", ARM.c_str());
521  geometry_msgs::Pose prev_block_pose;
522  // the block pose is saved since here below we search for
523  // a possibly closer block
524  prev_block_pose = block->getPose();
525  if (FSM::getClosestBlock(current_pose) < 0){
526  ROS_ERROR("BAXTER_FSM_%s: Error during PICK.", ARM.c_str());
527  next_state = ERR;
528  break;
529  }
530  /* If the block we initially went for is now farther then expected
531  go back into REACHing mode, setting the FSM to run autonomously
532  (since it won't receive a result from the simulation, which is
533  not carring out any trajectory).
534  */
535  if (dist3(prev_block_pose, block->getPose()) > maxInterBlockDist){
536  next_state = REACH;
537  block = nullptr;
538  break;
539  }
540 
541  next_state = FSM::pickBlock();
542 
543  if (next_state == ERR){
544  ROS_ERROR("BAXTER_FSM_%s: Error during PICK.", ARM.c_str());
545  break;
546  }
547 
548  auto_evolve = false;
549  break;
550  }
551 
552  case RAISE:
553  {
554  // raise the eef up of few cm
555  // next_state = obj.isBlue() ? PLACE_BLUE : REMOVE_RED
556 
557  ROS_INFO("BAXTER_FSM_%s: RAISE", ARM.c_str());
558  next_state = FSM::raiseBlock();
559 
560  if (next_state == ERR){
561  ROS_ERROR("BAXTER_FSM_%s: Error during PICK.", ARM.c_str());
562  break;
563  }
564  auto_evolve = false;
565  break;
566  }
567  case PLACE_BLUE:
568  {
569 
570  ROS_INFO("BAXTER_FSM_%s: PLACE_BLUE", ARM.c_str());
571  next_state = FSM::placeBlue();
572 
573  // implement a try-catch and throw exception from when block == nullptr
574  if (next_state == ERR){
575  ROS_ERROR("BAXTER_FSM_%s: Error during PLACE_BLUE", ARM.c_str());
576  break;
577  }
578 
579  ROS_INFO("\t BLOCK PLACED: %s", block->getName().c_str());
580  auto_evolve = false;
581 
582  BAXTER_ATTEMPTS_ = 0;
583  break;
584  }
585  case REMOVE_RED:
586  {
587  // retrieve empty position
588  // reach the position + offset
589 
590  ROS_INFO("BAXTER_FSM_%s: REMOVE_RED", ARM.c_str());
591  next_state = FSM::removeRed();
592 
593  if (next_state == ERR){
594  ROS_ERROR("BAXTER_FSM_%s: Error during REMOVE_RED", ARM.c_str());
595  break;
596  }
597  // plan and publish it
598  ROS_INFO("\t BLOCK MOVED: %s", block->getName().c_str());
599  auto_evolve = false;
600 
601  BAXTER_ATTEMPTS_ = 0;
602  break;
603  }
604  case END:
605  {
606  // Do nothing for now
607  // Should it (indirectly) terminate the node?
608  ROS_INFO("BAXTER_FSM_%s: END", ARM.c_str());
609  next_state = END;
610  ros::Duration(5.0).sleep();
611  break;
612  }
613  case ERR:
614  {
615  ROS_INFO("BAXTER_FSM_%s: ERR", ARM.c_str());
616  // in case of error perform in the same way as for IDLE
617  }
618  case IDLE:
619  {
620  ROS_WARN("BAXTER_FSM_%s: Going to rest state (%d)", ARM.c_str(), BAXTER_ATTEMPTS_);
621  next_state = FSM::rest();
622 
623  if (next_state == IDLE){ auto_evolve = false; }
624  // auto_evolve = true; //< already true
625  break;
626  }
627  }
628 
629  return auto_evolve;
630 }
